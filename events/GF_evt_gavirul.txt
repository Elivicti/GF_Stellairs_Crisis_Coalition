	is_triggered_only = yes

	picture = GFX_evt_GF_gavirul

	option = {
		name = GF_gavirul.1.a

				custom_tooltip = GF_gavirul.1.a.desc
			}
		}
		hidden_effect = {
			remove_country_flag = flag_GF_gavirul_project_started

			}

}

country_event = {
	id = GF_gavirul.2
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		set_country_flag = flag_GF_gavirul_project_finished
		every_owned_species = {
			limit = { has_trait = GF_trait_sequela }
			save_event_target_as = changing_species

			modify_species = {
				species = this
				base = this
				remove_trait = GF_trait_sequela
				change_scoped_species = yes
				add_traits_at_start_of_list = yes
				add_trait = trait_GF_sequela_immune

				effect = {
					save_event_target_as = new_main_species
				}
			}

			root = {
				every_owned_pop = {
					limit = { is_exact_same_species = event_target:changing_species }
					change_species = event_target:new_main_species #last_created_species
				}
	
				every_owned_leader = {
					limit = { is_exact_same_species = event_target:changing_species }
					change_species = event_target:new_main_species #last_created_species
				}
				every_pool_leader = {
					limit = { is_exact_same_species = event_target:changing_species }
					change_species = event_target:new_main_species #last_created_species
				}
				every_owned_army = {
					limit = {
						exists = species
						is_exact_same_species = event_target:changing_species
					}
					change_species = event_target:new_main_species #last_created_species
				}
				every_owned_ship = {
					limit = {
						is_ship_class = shipclass_colonizer
						is_exact_same_species = event_target:changing_species
					}
					change_species = event_target:new_main_species #last_created_species
				}
			}

		}
	}
}
