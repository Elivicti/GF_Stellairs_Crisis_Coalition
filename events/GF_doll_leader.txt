namespace = GF_doll_leader

country_event = {
	id = GF_doll_leader.0
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		from = {
			species = { has_trait = GF_trait_doll }
		}
	}

	immediate = {
		from = {
			add_trait = GF_trait_leader_doll
			switch = {
				trigger = leader_class
				admiral = {
					if = {
						limit = { species = { has_trait = GF_trait_tacticalchip } }
						add_trait = GF_trait_leader_admiraltrait_plus
					}
				}
				general = {
					if = {
						limit = { species = { has_trait = GF_trait_tacticalchip } }
						add_trait = GF_trait_leader_generaltrait_plus
					}
				}
				scientist = {
					if = {
						limit = { species = { has_trait = GF_trait_logical_subroutine } }
						add_trait = GF_trait_leader_scientisttrait_plus
					}
				}
				governor = {
					if = {
						limit = { species = { has_trait = GF_trait_zener_protocol } }
						add_trait = GF_trait_leader_governortrait_plus
					}
				}
			}
		}
	}
}
