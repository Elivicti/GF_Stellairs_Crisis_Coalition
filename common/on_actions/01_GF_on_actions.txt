# Executes after a policy has been changed
# use last_changed_policy to identify which policy it was
# This = Country
on_policy_changed = {
	events = {
		# 权利政策
		GF_doll.1110	# 完整权利 或 有限权利 - 可以生成领袖
		GF_doll.1120	# 侍从权利 或 商品权利 - 不可生成领袖

		# 任务分配政策
		GF_doll.1200	#
	}
}
