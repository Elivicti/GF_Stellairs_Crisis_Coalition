		}

		modifier = {
			GF_ANDROID_pop_happiness = 0.05
		}

		policy_flags = { GF_doll_work_policy_high }
		AI_weight = { weight = 1 }
	}
	option = {
		name = "GF_doll_work_policy_middle"

		valid = {
			custom_tooltip = {
				fail_text = GF_doll_work_policy_doll_has_full_right

			}
			OR = {
				has_policy_flag = GF_doll_right_limited
